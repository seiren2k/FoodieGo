142 private function findCodeUnits(string $filename): CodeUnitFindingVisitor
143 {
144 $nodes = (new ParserFactory)->createForHostVersion()->parse(
145 file_get_contents($filename),
146 );
147
148 assert($nodes !== null);
149
150 $traverser = new NodeTraverser;
151 $codeUnitFindingVisitor = new CodeUnitFindingVisitor;
152
153 $traverser->addVisitor(new NameResolver);
154 $traverser->addVisitor(new ParentConnectingVisitor);
155 $traverser->addVisitor($codeUnitFindingVisitor);
156
157 /* @noinspection UnusedFunctionResultInspection */
158 $traverser->traverse($nodes);
159
160 return $codeUnitFindingVisitor;
161 }
