62 public function forClass(string $className): MetadataCollection
63 {
64 assert(class_exists($className));
65
66 $result = [];
67
68 foreach (AnnotationRegistry::getInstance()->forClassName($className)->symbolAnnotations() as $annotation => $values) {
69 switch ($annotation) {
70 case 'backupGlobals':
71 $result[] = Metadata::backupGlobalsOnClass($this->stringToBool($values[0]));
72
73 break;
74
75 case 'backupStaticAttributes':
76 case 'backupStaticProperties':
77 $result[] = Metadata::backupStaticPropertiesOnClass($this->stringToBool($values[0]));
78
79 break;
80
81 case 'covers':
82 foreach ($values as $value) {
83 $value = $this->cleanUpCoversOrUsesTarget($value);
84
85 $result[] = Metadata::coversOnClass($value);

87
88 break;
89
90 case 'coversDefaultClass':
91 foreach ($values as $value) {
92 $result[] = Metadata::coversDefaultClass($value);
93 }
94
95 break;
96
97 case 'coversNothing':
98 $result[] = Metadata::coversNothingOnClass();
99
100 break;
101
102 case 'doesNotPerformAssertions':
103 $result[] = Metadata::doesNotPerformAssertionsOnClass();
104
105 break;
106
107 case 'group':
108 case 'ticket':
109 foreach ($values as $value) {
110 $result[] = Metadata::groupOnClass($value);
111 }
112
113 break;
114
115 case 'large':
116 $result[] = Metadata::groupOnClass('large');
117
118 break;
119
120 case 'medium':
121 $result[] = Metadata::groupOnClass('medium');
122
123 break;
124
125 case 'preserveGlobalState':
126 $result[] = Metadata::preserveGlobalStateOnClass($this->stringToBool($values[0]));
127
128 break;
129
130 case 'runClassInSeparateProcess':

132
133 break;
134
135 case 'runTestsInSeparateProcesses':

137
138 break;
139
140 case 'small':

142
143 break;
144
145 case 'testdox':
146 $result[] = Metadata::testDoxOnClass($values[0]);
147
148 break;

150 case 'uses':

152 $value = $this->cleanUpCoversOrUsesTarget($value);
153
154 $result[] = Metadata::usesOnClass($value);
155 }

157 break;

210 public function forMethod(string $className, string $methodName): MetadataCollection
211 {
212 assert(class_exists($className));
213 assert(method_exists($className, $methodName));
214
215 $result = [];
216
217 foreach (AnnotationRegistry::getInstance()->forMethod($className, $methodName)->symbolAnnotations() as $annotation => $values) {
218 switch ($annotation) {
219 case 'after':
220 $result[] = Metadata::after(0);
221
222 break;
223
224 case 'afterClass':
225 $result[] = Metadata::afterClass(0);
226
227 break;
228
229 case 'backupGlobals':
230 $result[] = Metadata::backupGlobalsOnMethod($this->stringToBool($values[0]));
231
232 break;
233
234 case 'backupStaticAttributes':
235 case 'backupStaticProperties':
236 $result[] = Metadata::backupStaticPropertiesOnMethod($this->stringToBool($values[0]));
237
238 break;
239
240 case 'before':
241 $result[] = Metadata::before(0);
242
243 break;
244
245 case 'beforeClass':
246 $result[] = Metadata::beforeClass(0);
247
248 break;
249
250 case 'covers':
251 foreach ($values as $value) {
252 $value = $this->cleanUpCoversOrUsesTarget($value);
253
254 $result[] = Metadata::coversOnMethod($value);
255 }
256
257 break;
258
259 case 'coversNothing':
260 $result[] = Metadata::coversNothingOnMethod();
261
262 break;
263
264 case 'dataProvider':
265 foreach ($values as $value) {
266 $value = rtrim($value, " ()\n\r\t\v\x00");
267
268 if (str_contains($value, '::')) {
269 $result[] = Metadata::dataProvider(...explode('::', $value));
270
271 continue;
272 }
273
274 $result[] = Metadata::dataProvider($className, $value);
275 }
276
277 break;
278
279 case 'depends':
280 foreach ($values as $value) {
281 $deepClone = false;
282 $shallowClone = false;
283
284 if (str_starts_with($value, 'clone ')) {
285 $deepClone = true;
286 $value = substr($value, strlen('clone '));
287 } elseif (str_starts_with($value, '!clone ')) {
288 $value = substr($value, strlen('!clone '));
289 } elseif (str_starts_with($value, 'shallowClone ')) {
290 $shallowClone = true;
291 $value = substr($value, strlen('shallowClone '));
292 } elseif (str_starts_with($value, '!shallowClone ')) {
293 $value = substr($value, strlen('!shallowClone '));
294 }
295
296 if (str_contains($value, '::')) {
297 [$_className, $_methodName] = explode('::', $value);
298
299 assert($_className !== '');
300 assert($_methodName !== '');
301
302 if ($_methodName === 'class') {
303 $result[] = Metadata::dependsOnClass($_className, $deepClone, $shallowClone);
304
305 continue;
306 }
307
308 $result[] = Metadata::dependsOnMethod($_className, $_methodName, $deepClone, $shallowClone);
309
310 continue;
311 }
312
313 $result[] = Metadata::dependsOnMethod($className, $value, $deepClone, $shallowClone);
314 }
315
316 break;
317
318 case 'doesNotPerformAssertions':
319 $result[] = Metadata::doesNotPerformAssertionsOnMethod();
320
321 break;
322
323 case 'excludeGlobalVariableFromBackup':
324 foreach ($values as $value) {
325 $result[] = Metadata::excludeGlobalVariableFromBackupOnMethod($value);
326 }
327
328 break;
329
330 case 'excludeStaticPropertyFromBackup':
331 foreach ($values as $value) {
332 $tmp = explode(' ', $value);
333
334 if (count($tmp) !== 2) {
335 continue;
336 }
337
338 $result[] = Metadata::excludeStaticPropertyFromBackupOnMethod(
339 trim($tmp[0]),
340 trim($tmp[1]),
341 );
342 }
343
344 break;
345
346 case 'group':
347 case 'ticket':
348 foreach ($values as $value) {
349 $result[] = Metadata::groupOnMethod($value);
350 }
351
352 break;
353
354 case 'large':
355 $result[] = Metadata::groupOnMethod('large');
356
357 break;
358
359 case 'medium':
360 $result[] = Metadata::groupOnMethod('medium');
361
362 break;
363
364 case 'postCondition':
365 $result[] = Metadata::postCondition(0);
366
367 break;
368
369 case 'preCondition':
370 $result[] = Metadata::preCondition(0);
371
372 break;
373
374 case 'preserveGlobalState':
375 $result[] = Metadata::preserveGlobalStateOnMethod($this->stringToBool($values[0]));
376
377 break;
378
379 case 'runInSeparateProcess':
380 $result[] = Metadata::runInSeparateProcess();
381
382 break;
383
384 case 'small':
385 $result[] = Metadata::groupOnMethod('small');
386
387 break;
388
389 case 'test':
390 $result[] = Metadata::test();
391
392 break;
393
394 case 'testdox':
395 $result[] = Metadata::testDoxOnMethod($values[0]);
396
397 break;
398
399 case 'uses':
400 foreach ($values as $value) {
401 $value = $this->cleanUpCoversOrUsesTarget($value);
402
403 $result[] = Metadata::usesOnMethod($value);
404 }
405
406 break;
407 }
408 }
409
410 if (method_exists($className, $methodName)) {
411 try {
412 $result = array_merge(
413 $result,
414 $this->parseRequirements(
415 AnnotationRegistry::getInstance()->forMethod($className, $methodName)->requirements(),
416 'method',
417 ),
418 );
419 } catch (InvalidVersionRequirementException $e) {
420 EventFacade::emitter()->testRunnerTriggeredWarning(
421 sprintf(
422 'Method %s::%s is annotated using an invalid version requirement: %s',
423 $className,
424 $methodName,
425 $e->getMessage(),
426 ),
427 );
428 }
429 }
430
431 if (!empty($result) &&
432 !isset(self::$deprecationEmittedForMethod[$className . '::' . $methodName]) &&
433 !str_starts_with($className, 'PHPUnit\TestFixture')) {
434 EventFacade::emitter()->testRunnerTriggeredDeprecation(
435 sprintf(
436 'Metadata found in doc-comment for method %s::%s(). Metadata in doc-comments is deprecated and will no longer be supported in PHPUnit 12. Update your test code to use attributes instead.',
437 $className,
438 $methodName,
439 ),
440 );
441
442 self::$deprecationEmittedForMethod[$className . '::' . $methodName] = true;
443 }
444
445 return MetadataCollection::fromArray($result);

486 private function parseRequirements(array $requirements, string $level): array
487 {
488 $result = [];
489
490 if (!empty($requirements['PHP'])) {
491 $versionRequirement = new ComparisonRequirement(
492 $requirements['PHP']['version'],
493 new VersionComparisonOperator(empty($requirements['PHP']['operator']) ? '>=' : $requirements['PHP']['operator']),
494 );
495
496 if ($level === 'class') {
497 $result[] = Metadata::requiresPhpOnClass($versionRequirement);
498 } else {
499 $result[] = Metadata::requiresPhpOnMethod($versionRequirement);
500 }
501 } elseif (!empty($requirements['PHP_constraint'])) {
502 $versionRequirement = new ConstraintRequirement($requirements['PHP_constraint']['constraint']);
503
504 if ($level === 'class') {
505 $result[] = Metadata::requiresPhpOnClass($versionRequirement);
506 } else {
507 $result[] = Metadata::requiresPhpOnMethod($versionRequirement);
508 }
509 }
510
511 if (!empty($requirements['extensions'])) {
512 foreach ($requirements['extensions'] as $extension) {
513 if (isset($requirements['extension_versions'][$extension])) {
514 continue;
515 }
516
517 if ($level === 'class') {
518 $result[] = Metadata::requiresPhpExtensionOnClass($extension, null);
519 } else {
520 $result[] = Metadata::requiresPhpExtensionOnMethod($extension, null);
521 }
522 }
523 }
524
525 if (!empty($requirements['extension_versions'])) {
526 foreach ($requirements['extension_versions'] as $extension => $version) {
527 $versionRequirement = new ComparisonRequirement(
528 $version['version'],
529 new VersionComparisonOperator(empty($version['operator']) ? '>=' : $version['operator']),
530 );
531
532 if ($level === 'class') {
533 $result[] = Metadata::requiresPhpExtensionOnClass($extension, $versionRequirement);
534 } else {
535 $result[] = Metadata::requiresPhpExtensionOnMethod($extension, $versionRequirement);
536 }
537 }
538 }
539
540 if (!empty($requirements['PHPUnit'])) {
541 $versionRequirement = new ComparisonRequirement(
542 $requirements['PHPUnit']['version'],
543 new VersionComparisonOperator(empty($requirements['PHPUnit']['operator']) ? '>=' : $requirements['PHPUnit']['operator']),
544 );
545
546 if ($level === 'class') {
547 $result[] = Metadata::requiresPhpunitOnClass($versionRequirement);
548 } else {
549 $result[] = Metadata::requiresPhpunitOnMethod($versionRequirement);
550 }
551 } elseif (!empty($requirements['PHPUnit_constraint'])) {
552 $versionRequirement = new ConstraintRequirement($requirements['PHPUnit_constraint']['constraint']);
553
554 if ($level === 'class') {
555 $result[] = Metadata::requiresPhpunitOnClass($versionRequirement);
556 } else {
557 $result[] = Metadata::requiresPhpunitOnMethod($versionRequirement);
558 }
559 }
560
561 if (!empty($requirements['OSFAMILY'])) {
562 if ($level === 'class') {
563 $result[] = Metadata::requiresOperatingSystemFamilyOnClass($requirements['OSFAMILY']);
564 } else {
565 $result[] = Metadata::requiresOperatingSystemFamilyOnMethod($requirements['OSFAMILY']);
566 }
567 }
568
569 if (!empty($requirements['OS'])) {
570 if ($level === 'class') {
571 $result[] = Metadata::requiresOperatingSystemOnClass($requirements['OS']);
572 } else {
573 $result[] = Metadata::requiresOperatingSystemOnMethod($requirements['OS']);
574 }
575 }
576
577 if (!empty($requirements['functions'])) {
578 foreach ($requirements['functions'] as $function) {
579 $pieces = explode('::', $function);
580
581 if (count($pieces) === 2) {
582 if ($level === 'class') {
583 $result[] = Metadata::requiresMethodOnClass($pieces[0], $pieces[1]);
584 } else {
585 $result[] = Metadata::requiresMethodOnMethod($pieces[0], $pieces[1]);
586 }
587 } elseif ($level === 'class') {
588 $result[] = Metadata::requiresFunctionOnClass($function);
589 } else {
590 $result[] = Metadata::requiresFunctionOnMethod($function);
591 }
592 }
593 }
594
595 if (!empty($requirements['setting'])) {
596 foreach ($requirements['setting'] as $setting => $value) {
597 if ($level === 'class') {
598 $result[] = Metadata::requiresSettingOnClass($setting, $value);
599 } else {
600 $result[] = Metadata::requiresSettingOnMethod($setting, $value);
601 }
602 }
603 }
604
605 return $result;
606 }
